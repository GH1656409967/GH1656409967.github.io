 1using namespace std;
 2typedef long long ll;
 3const int inf = 0x3f3f3f3f;
 4const int N = 2e4+5;
 5const int M = 5e4+5;
 6
 7struct E {
 8    int to, next;
 9} e[M];
10
11int H[N], tot;
12
13int S[N], top;
14int dfn[N], low[N], bel[N], idx, scc;
15
16int n, m, cost[N];
17int u, v;
18
19int sc[N], id[N];
20
21void add(int from, int to) {
22    e[tot] = {to, H[from]};
23    H[from] = tot++;
24}

26void dfs(int u) {
27    dfn[u] = low[u] = ++idx;
28    S[++top]=u;
29
30    for(int i=H[u]; ~i; i=e[i].next) {
31        int v = e[i].to;
32        if(!dfn[v]) {
33            dfs(v);
34            low[u] = min(low[u], low[v]);
35        } else if(!bel[v])
36            low[u] = min(low[u], dfn[v]);
37    }
38    if(low[u]==dfn[u]) {
39        scc++;
40        int t;
41        do {
42            t=S[top--];
43            bel[t]=scc;
44        } while(t!=u);
45    }
46}

48void tarjan() {
49    memset(dfn, 0, sizeof(dfn));
50    memset(bel, 0, sizeof(bel));
51    idx = scc = top = 0;
52    for(int i=1; i<=n; i++) {
53        if(!dfn[i])
54            dfs(i);
55    }
56}

58void solve(int &num, int &sum) {
59    memset(sc, inf, sizeof(sc));
60    memset(id, 0, sizeof(id));
61
62    for(int i=1; i<=n; i++)
63        sc[bel[i]] = min(sc[bel[i]], cost[i]);
64
65    for(int u=1; u<=n; u++) {
66        for(int i=H[u]; ~i; i=e[i].next) {
67            int v = e[i].to;
68            if(bel[u]!=bel[v]) {
69                id[bel[v]]++;
70            }
71        }
72    }
73
74    int a=0, b=0;
75    for(int i=1; i<=scc; i++) {
76        if(!id[i])
77            num++, sum+=sc[i];
78    }
79}
